C++ Final Project

For my final project in C++, I wrote a program that reads exam scores from a file, calculates the average, determines how many scores are above and below average, sorts them, and writes the results to a new file. It also identifies the highest and lowest scores in the set. 

#include <iostream>
#include <fstream>
using namespace std;

void selectionSort(double scores[], int n);

int main() {
    int numel;
    double sum = 0, avg, aAvg = 0, bAvg = 0;

    ifstream fin;
    fin.open("scores.dat");
    cout << "Enter the number of scores in the file: ";
    cin >> numel;
    double scores[numel];

    for (int i = 0; i < numel; i++) {
        fin >> scores[i];
        sum += scores[i];
    }
    fin.close();
    avg = sum / numel;

    for (int i = 0; i < numel; i++) {
        if (scores[i] > avg) aAvg++;
        if (scores[i] < avg) bAvg++;
    }

    int n = sizeof(scores)/sizeof(scores[0]);
    selectionSort(scores, n);

    ofstream fout;
    fout.open("averageInfo.dat");
    fout << "The sorted scores are:" << endl;
    for (int i = 0; i < n; i++) {
        fout << scores[i] << endl;
    }
    fout << "The average of the scores is: " << avg << endl;
    fout << "The lowest score is: " << scores[0] << endl;
    fout << "The highest score is: " << scores[numel - 1] << endl;
    fout << "There are " << bAvg << " scores below the average." << endl;
    fout << "There are " << aAvg << " scores above the average." << endl;
    fout.close();

    cout << "Done processing. You will find the results in averageInfo.dat." << endl;
    return 0;
}

void selectionSort(double arr[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                double temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}
